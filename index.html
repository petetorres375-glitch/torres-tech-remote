import os
from datetime import date
 
folder = "/reports"
today = date.today().strftime("%Y-%m-%d")
 
for fname in os.listdir(folder):
  if not fname.startswith(today):
    new = f"{today}_{fname}"
    os.rename(
      os.path.join(folder, fname),
      os.path.join(folder, new))
 
print("Done — files renamed.")
